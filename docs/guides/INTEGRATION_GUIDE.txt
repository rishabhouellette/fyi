=============================================

    """Refresh all module data"""
    try:
        if hasattr(self, 'calendar_frame'):
            self.calendar_frame.refresh()
        if hasattr(self, 'analytics_frame'):
            self.analytics_frame.refresh()
        if hasattr(self, 'inbox_frame'):
            self.inbox_frame.refresh()
        if hasattr(self, 'library_frame'):
            self.library_frame.refresh()
        if hasattr(self, 'team_frame'):
            self.team_frame.refresh()
        if hasattr(self, 'fc_frame'):
            self.fc_frame.refresh()
        if hasattr(self, 'hashtag_frame'):
            self.hashtag_frame.refresh()
    except Exception as e:
        logger.error(f"Refresh error: {e}")

TEAM_ID PARAMETER
=================

Each module takes a team_id parameter:
- Currently set to team_id=1 (single team)
- For multi-tenant: pass dynamic team_id from user context
- Each team has isolated data in the database

EXAMPLE FULL INTEGRATION
========================

Here's a minimal working example for integration:

```python
import customtkinter as ctk
from database_manager import get_db_manager
from ui_calendar import ContentCalendarFrame
from ui_analytics import AnalyticsDashboardFrame
from ui_bulk_upload import BulkUploadFrame
from ui_social_inbox import SocialInboxFrame
from ui_content_library import ContentLibraryFrame
from ui_team_collaboration import TeamCollaborationFrame
from ui_first_comments import FirstCommentsFrame
from ui_hashtag_tools import HashtagToolsFrame

class MainApp(ctk.CTk):
    def __init__(self):
        super().__init__()
        
        # Initialize database
        self.db = get_db_manager()
        self.team_id = 1  # Default to team 1
        
        # Create main UI
        self.title("FYI Social Media Manager")
        self.geometry("1200x700")
        
        # Create tabview
        self.tabview = ctk.CTkTabview(self)
        self.tabview.pack(fill="both", expand=True, padx=10, pady=10)
        
        # Add Phase 1 tabs
        cal_tab = self.tabview.add("Calendar")
        ContentCalendarFrame(cal_tab, team_id=self.team_id).pack(fill="both", expand=True)
        
        ana_tab = self.tabview.add("Analytics")
        AnalyticsDashboardFrame(ana_tab, team_id=self.team_id).pack(fill="both", expand=True)
        
        bulk_tab = self.tabview.add("Bulk Upload")
        BulkUploadFrame(bulk_tab, team_id=self.team_id).pack(fill="both", expand=True)
        
        # Add Phase 2 tabs
        inbox_tab = self.tabview.add("Inbox")
        SocialInboxFrame(inbox_tab, team_id=self.team_id).pack(fill="both", expand=True)
        
        lib_tab = self.tabview.add("Content Library")
        ContentLibraryFrame(lib_tab, team_id=self.team_id).pack(fill="both", expand=True)
        
        team_tab = self.tabview.add("Team")
        TeamCollaborationFrame(team_tab, team_id=self.team_id).pack(fill="both", expand=True)
        
        fc_tab = self.tabview.add("First Comments")
        FirstCommentsFrame(fc_tab, team_id=self.team_id).pack(fill="both", expand=True)
        
        ht_tab = self.tabview.add("Hashtag Tools")
        HashtagToolsFrame(ht_tab, team_id=self.team_id).pack(fill="both", expand=True)

if __name__ == "__main__":
    app = MainApp()
    app.mainloop()
```

TROUBLESHOOTING
===============

Import Error: 
  - Ensure all ui_*.py files are in same directory as main.py
  - Check Python path includes current directory

Database Error:
  - Run: python -c "from database_manager import get_db_manager; db = get_db_manager()"
  - Check data/ folder exists and is writable
  - Check database_schema.sql was executed

Module Not Showing:
  - Check tab name matches module name
  - Verify frame.pack() is called
  - Check for exceptions in logs
